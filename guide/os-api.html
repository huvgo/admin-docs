后端项目 #

导入项目 #

  • 首先导入项目到你的开发工具中
  • 修改配置信息
// 导入项目成功后

// 修改数据库连接信息
spring.datasource.url=jdbc:mysql://localhost:3306/admin_os?useUnicode=true&useSSL=false&characterEncoding=utf8
spring.datasource.username=root
spring.datasource.password=root
// 修改redis连接信息
spring.redis.host=127.0.0.1
spring.redis.port=6379
......

运行项目 #

// 运行AdminOsApplication.java
@SpringBootApplication
public class AdminOsApplication {

    public static void main(String[] args) {
        SpringApplication.run(AdminOsApplication.class, args);
    }

}
